            ```python
            def generate_srt_from_words(words, max_line_length=42, max_words_per_line=7):
                """
                Generates a valid SRT formatted string from Whisper's word-level timestamp data.

                Args:
                    words (list): The list of TranscriptionWord objects from the Whisper API response.
                    max_line_length (int): The approximate maximum number of characters per subtitle line.
                    max_words_per_line (int): The maximum number of words to group into a single line.
                """
                print("DEBUG: Generating SRT from word-level timestamps.", file=sys.stderr)
                srt_blocks = []
                current_block_text = []
                start_time = None
                end_time = None
                cue_counter = 1

                if not words:
                    print("WARNING: Word list for SRT generation is empty.", file=sys.stderr)
                    return ""

                for i, word_data in enumerate(words):
                    word = word_data.word
                    
                    if start_time is None:
                        start_time = word_data.start

                    current_block_text.append(word)
                    end_time = word_data.end
                    
                    is_last_word = (i == len(words) - 1)
                    current_line = " ".join(current_block_text)
                    
                    # Determine if the block should end
                    end_of_block = False
                    if is_last_word:
                        end_of_block = True
                    elif len(current_line) >= max_line_length:
                        end_of_block = True
                    elif len(current_block_text) >= max_words_per_line:
                        end_of_block = True
                    elif (words[i+1].start - end_time) > 1.0: # Check for a pause of >1s
                        end_of_block = True

                    if end_of_block:
                        start_h, rem = divmod(start_time, 3600)
                        start_m, start_s_full = divmod(rem, 60)
                        start_s = int(start_s_full)
                        start_ms = int((start_s_full - start_s) * 1000)
                        
                        end_h, rem = divmod(end_time, 3600)
                        end_m, end_s_full = divmod(rem, 60)
                        end_s = int(end_s_full)
                        end_ms = int((end_s_full - end_s) * 1000)
                        
                        timestamp_start = f"{int(start_h):02}:{int(start_m):02}:{int(start_s):02},{start_ms:03}"
                        timestamp_end = f"{int(end_h):02}:{int(end_m):02}:{int(end_s):02},{end_ms:03}"
                        
                        srt_blocks.append(str(cue_counter))
                        srt_blocks.append(f"{timestamp_start} --> {timestamp_end}")
                        srt_blocks.append(current_line)
                        srt_blocks.append("")
                        
                        cue_counter += 1
                        current_block_text = []
                        start_time = None
                
                print(f"DEBUG: Generated {cue_counter - 1} SRT blocks.", file=sys.stderr)
                return "\\n".join(srt_blocks)
            ```

       ```python
       def save_wave_file(path: str, audio_data: bytes, channels=2, sample_rate=48000, sample_width_bytes=2):
           """Saves raw PCM audio data to a WAV file."""
           print(f"DEBUG: Saving {len(audio_data)} bytes of audio data to {path}", file=sys.stderr)
           try:

                   wf.setnchannels(channels)
                   wf.setsampwidth(sample_width_bytes)
                   wf.setframerate(sample_rate)

               return path
           except Exception as e:

       async def receive_audio_chunks(session, audio_chunk_collector):
           """Background task to collect incoming audio chunks from Lyria."""
           print("DEBUG: Lyria: Listening for audio chunks...", file=sys.stderr)
           try:
               async for message in session.receive():
                   if message.server_content and message.server_content.audio_chunks:
                       audio_data = message.server_content.audio_chunks[0].data
                       audio_chunk_collector.append(audio_data)
                   elif message.server_content and message.server_content.filtered_prompt:
                       print(f"WARNING: Lyria prompt filtered: {message.server_content.filtered_prompt}", file=sys.stderr)
               await asyncio.sleep(1e-9)
           except asyncio.CancelledError:
               print("DEBUG: Lyria: Audio receiving task cancelled.", file=sys.stderr)
           except Exception as e:
               print(f"ERROR: Lyria: Error in receiving audio chunks: {e}", file=sys.stderr)

       async def _run_lyria_session_async(prompt: str, duration_seconds: float):
           """Connects to Lyria, generates music, and returns the audio data."""
           print(f"DEBUG: Starting Lyria session for {duration_seconds}s with prompt: '{prompt}'", file=sys.stderr)
           
           # CRITICAL: To use the Lyria (Gemini) endpoint instead of the default Vertex AI endpoint,
           # we temporarily unset the environment variables that force the client into Vertex AI mode.
           # This allows genai.Client() to use Application Default Credentials for the global Gemini API.
           client = None
           original_env = {
               'GOOGLE_GENAI_USE_VERTEXAI': os.environ.pop('GOOGLE_GENAI_USE_VERTEXAI', None),
               'GOOGLE_CLOUD_PROJECT': os.environ.pop('GOOGLE_CLOUD_PROJECT', None),
               'GOOGLE_CLOUD_LOCATION': os.environ.pop('GOOGLE_CLOUD_LOCATION', None)
           }

           try:
               # With the Vertex env vars removed, the client will use the global endpoint with ADC.
               # The 'v1alpha' API version is required for Lyria's streaming API.
               client = genai.Client(http_options={'api_version': 'v1alpha'})
           except Exception as e:
               print(f"ERROR: Failed to initialize Gemini Client for Lyria: {e}", file=sys.stderr)
               # Restore environment variables in case of failure
               for key, value in original_env.items():
                   if value is not None:
                       os.environ[key] = value
               raise

           audio_chunk_collector = []
           
           try:
               async with client.aio.live.music.connect(model='models/lyria-realtime-exp') as session:
                   print("DEBUG: Lyria session connected.", file=sys.stderr)
                   async with asyncio.TaskGroup() as tg:
                       audio_receive_task = tg.create_task(receive_audio_chunks(session, audio_chunk_collector))
                       
                       await session.set_weighted_prompts(
                           prompts=[types.WeightedPrompt(text=prompt, weight=1.0)]
                       )
                       await session.set_music_generation_config(
                           config=types.LiveMusicGenerationConfig(bpm=90, guidance=4.0)
                       )
                       
                       print("DEBUG: Lyria: Starting music playback...", file=sys.stderr)
                       await session.play()
                       await asyncio.sleep(duration_seconds)
                       print("DEBUG: Lyria: Stopping music playback...", file=sys.stderr)
                       await session.stop()
                       
                       await asyncio.sleep(1) # Give a moment for final chunks to arrive
                       audio_receive_task.cancel()

           except Exception as e:
               print(f"ERROR: Lyria session failed: {e}", file=sys.stderr)
               return None
           finally:
                # IMPORTANT: Restore the environment variables so other parts of the app work correctly.
                for key, value in original_env.items():
                    if value is not None:
                        os.environ[key] = value
           
           return b''.join(audio_chunk_collector)

               return None
           
           return save_wave_file(output_path, audio_data)
       ```

     * **Polling, Downloading, and Processing (MANDATORY for ALL Veo operations on Vertex AI)**:
       - **CRITICAL**: In a Vertex AI environment, Veo saves the video to a Google Cloud Storage (GCS) URI that you must provide. The following workflow is mandatory.

       - **Helper Function for GCS Download**: First, ensure you have a helper function to download files from GCS. `gcloud storage` is the modern, recommended tool for this.
         ```python
         import subprocess
         import re

         def download_from_gcs(gcs_uri: str, local_dir: str) -> Optional[str]:
             """Downloads content from a GCS URI to a local directory using 'gcloud storage cp'."""
             if not gcs_uri.startswith("gs://"):
                 print(f"ERROR: Invalid GCS URI: {gcs_uri}", file=sys.stderr)
                 return None
             
             # The destination path for the download
             local_path = os.path.join(local_dir, os.path.basename(gcs_uri.strip('/')))
             
             try:
                 # Use --recursive for directory URIs, which Veo might return.
                 recursive_flag = ["--recursive"] if gcs_uri.endswith('/') else []
                 command = ["gcloud", "storage", "cp"] + recursive_flag + [gcs_uri, local_dir]

                 print(f"DEBUG: Downloading from {gcs_uri} to {local_dir} using 'gcloud storage'...", file=sys.stderr)
                 print(f"DEBUG: Executing command: {' '.join(command)}", file=sys.stderr)
                 
                 subprocess.run(command, check=True, capture_output=True, text=True)
                 
                 # After download, find the actual video file, as gcloud might create subdirectories.
                 downloaded_mp4_files = []
                 for root, _, files in os.walk(local_dir):
                     for name in files:
                         if name.lower().endswith('.mp4'):
                             downloaded_mp4_files.append(os.path.join(root, name))
                 
                 if not downloaded_mp4_files:
                     raise FileNotFoundError("gcloud storage cp ran, but no MP4 file was found in the destination.")

                 # Assume the largest MP4 file is the one we want.
                 main_video_file = max(downloaded_mp4_files, key=os.path.getsize)

                 if os.path.exists(main_video_file) and os.path.getsize(main_video_file) > 0:
                     print(f"DEBUG: GCS content downloaded successfully. Main video file: {main_video_file}", file=sys.stderr)
                     return main_video_file
                 else:
                     raise FileNotFoundError("GCS download completed but the main video file is missing or empty.")
         ```
